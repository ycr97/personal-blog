##### 定义:
线性表的顺序表示指的是用一组地址连续的存储单元依次存储线性表的数据元素.
结构如下:
![线性表顺序表存储结构示意图](/personal-blog/images/csdn/115297131/01-8fdbfb988c0c.png)
使用java实现类似ArrayList原理的简单线性表的顺序结构:

```
/**
 * @author ycr
 * @date 2021/3/29
 */
@SuppressWarnings("all ")
public class ArrayListDemo<E> {

    /*初始容量*/
    private static final int DEFAULT_CAPACITY = 10;

    /*存放数据的数组*/
    private Object[] elementData;

    /*当前长度*/
    private int size;

    /**
     * 有参构造,初始化数组长度
     *
     * @param initCapacity 初始容量
     */
    public ArrayListDemo(int initCapacity) {
        if (initCapacity > 0) {
            elementData = new Object[initCapacity];
        } else if (initCapacity == 0) {
            elementData = new Object[DEFAULT_CAPACITY];
        } else {
            throw new RuntimeException("初始容量不能小于0");
        }
    }

    public ArrayListDemo() {
        elementData = new Object[DEFAULT_CAPACITY];
    }

    /**
     * 添加元素到最后一个位置
     * 时间复杂度O(1)
     *
     * @param ele ele
     */
    public void add(E ele) {
        examinationCapacity(elementData.length);
        elementData[size++] = ele;
    }

    /**
     * 添加元素到指定索引位置,插入位置不为最后的话,
     * 该索引及其后面的元素整体后移以为,会检查是否扩容索引是否合法
     * 时间复杂度最坏O(n)
     *
     * @param index
     * @param ele
     */
    public void add(int index, E ele) {
        rangeCheck(index);
        examinationCapacity(elementData.length + 1);
        if (index < size) {
            System.arraycopy(elementData, index, elementData, index + 1, size - index);
            elementData[index] = ele;
            size++;
        } else {
            add(ele);
        }
    }

    /**
     * 扩容方法,数组拷贝到新的内存区域再将引用给elementData
     * 时间复杂度O(n)
     *
     * @param capacity
     */
    private void examinationCapacity(int capacity) {
        if (size == elementData.length) {
            elementData = Arrays.copyOf(elementData, capacity + (capacity >> 1));
        }
    }

    public E get(int index) {
        rangeCheck(index);
        return (E) elementData[index];
    }

    /**
     * 删除指定元素
     *
     * @param o
     */
    public void remove(Object o) {
        for (int i = 0; i < elementData.length; i++) {
            if (elementData[i].equals(o)) {
                remove(i);
            }
        }
    }

    /**
     * 删除指定索引元素
     * 时间复杂度最坏O(n)
     *
     * @param index
     */
    public void remove(int index) {
        rangeCheck(index);
        System.arraycopy(elementData, index + 1, elementData, index, size - index);
        size--;
    }

    /**
     * 检查索引是否合法
     *
     * @param index
     */
    private void rangeCheck(int index) {
        if (index < 0 || index > size) {
            throw new IndexOutOfBoundsException("索引越界!!!");
        }
    }

    public int size() {
        return size;
    }

    public boolean isEmpty() {
        return size == 0;
    }

    @Override
    public String toString() {
        StringBuilder builder = new StringBuilder();
        builder.append('[');
        for (int i = 0; i < size; i++) {
            builder.append(elementData[i].toString());
            if (i < size - 1) {
                builder.append(',');
            }
        }
        builder.append(']');
        return builder.toString();
    }

    public static void main(String[] args) {
        ArrayListDemo<String> demo = new ArrayListDemo<>();
        demo.add("a");
        demo.add("b");
        demo.add("c");
        demo.add("d");
        demo.add("e");
        demo.add("f");
        demo.add("g");
        demo.add("h");
        demo.add("i");
        demo.add("j");
        demo.add("k");
        demo.add("l");
        demo.add(10, "m");
        System.out.println(demo.size());
        demo.remove(1);
        System.out.println(demo);
        System.out.println(demo.size());
    }
}
```
