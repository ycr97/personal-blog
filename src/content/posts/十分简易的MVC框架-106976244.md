### 简易的MVC框架
模仿SpringMvc框架,其中核心思想是我们引入一个前端控制器;所谓前端控制器指的是所有请求均到达前端控制器,再由前端控制器分发请求到对应的映射处理器,由控制器处理完之后返回ModelAndView对象再由前端控制器统一设置共享数据到HttpRequest对象和请求转发到客户端.其中主要是两个注解@COntroller和@RrequestMapping需要由我们去扫描确定在哪些类哪些方法上面再使用反射调用方法处理对应请求资源的Controller中对应被@RequestMapping注解标注的方法.下面是具体代码

```java
package com.yy.easyMvc.util;

import org.junit.Test;
import java.io.File;
import java.lang.annotation.Annotation;
import java.net.JarURLConnection;
import java.net.URL;
import java.util.ArrayList;
import java.util.Enumeration;
import java.util.List;
import java.util.jar.JarEntry;
import java.util.jar.JarFile;

/**
 * @author ycr
 * @date 2020/6/24
 */
public class ClassUtil {

    /**
     * 获取指定包(包括子包)中指定注解的所有类并返回
     * @param packageName 包名
     * @param annotationClass 注解类型
     * @return list
     */
    public static List<Class<?>> getClassListByAnnotation(String packageName,
                                                          Class<? extends Annotation> annotationClass){

        List<Class<?>> classList = new ArrayList<>();
        try {
            Enumeration<URL> urls = Thread.currentThread().getContextClassLoader()
                    .getResources(packageName.replaceAll("\\.", "/"));
            while (urls.hasMoreElements()) {
                URL url = urls.nextElement();
                if (url != null) {
                    String protocol = url.getProtocol();
                    if ("file".equals(protocol)) {
                        String packagePath = url.getPath();
                        addClassByAnnotation(classList, packagePath, packageName, annotationClass);
                    }else if ("jar".equals(protocol)) {
                        JarURLConnection connection = (JarURLConnection) url.openConnection();
                        JarFile jarFile = connection.getJarFile();
                        Enumeration<JarEntry> jarEntries = jarFile.entries();
                        while (jarEntries.hasMoreElements()) {
                            JarEntry jarEntry = jarEntries.nextElement();
                            String jarEntryName = jarEntry.getName();
                            if (jarEntryName.endsWith(".class")) {
                                String className = jarEntryName.substring(0, jarEntryName.lastIndexOf("."))
                                        .replaceAll("/", ".");
                                Class<?> cls = Class.forName(className);
                                if (cls.isAnnotationPresent(annotationClass)) {
                                    classList.add(cls);
                                }
                            }

                        }
                    }
                }
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
        return classList;
    }

    /**
     *
     * @param classList  存储class的ArrayList
     * @param packagePath 包
     * @param packageName 包
     * @param annotationClass 注解类型
     */
    private static void addClassByAnnotation(List<Class<?>> classList, String packagePath,
                                      String packageName, Class<? extends Annotation> annotationClass){
        try {
            File[] files = getClassFiles(packagePath);
            if (files != null) {
                for (File file : files) {
                    String fileName = file.getName();
                    if (file.isFile()) {
                        String className = getClassName(packageName, fileName);
                        Class<?> cls = Class.forName(className);
                        if (cls.isAnnotationPresent(annotationClass)) {
                            classList.add(cls);
                        }
                    }else {
                        String subPackagePath = getSubPackagePath(packagePath, fileName);
                        String subPackageName = getSubPackagePath(packageName, fileName);
                        addClassByAnnotation(classList, subPackagePath, subPackageName, annotationClass);
                    }
                }
            }
        }catch (ClassNotFoundException e) {
            e.printStackTrace();
        }

    }

    /**
     *
     * @param packagePath 包路径
     * @return files
     */
    private static File[] getClassFiles(String packagePath) {
        return new File(packagePath).listFiles(
                file -> (file.isFile() && file.getName().endsWith(".class")) || file.isDirectory());
    }

    /**
     *
     * @param packageName 包名
     * @param fileName 文件名
     * @return className
     */
    private static String getClassName(String packageName, String fileName) {
        String className  = fileName.substring(0, fileName.lastIndexOf("."));
        if (hasLength(packageName)) {
            className = packageName + "." + className;
        }
        return className;
    }

    /**
     *
     * @param packagePath
     * @param filePath
     * @return
     */
    private static String getSubPackagePath(String packagePath, String filePath) {
        String subPackagePath = filePath;
        if (hasLength(subPackagePath)) {
            subPackagePath = packagePath + "/" + subPackagePath;
        }
        return subPackagePath;
    }

    private static String getSubPackageName(String packageName, String filePath) {
        String subPackageName = filePath;
        if (hasLength(packageName)) {
            subPackageName = packageName + "." + subPackageName;
        }
        return subPackageName;
    }

    public static boolean hasLength(String str) {
        return str != null && !"".equals(str.trim());
    }

    public static boolean isEmpty(String str) {
        return !hasLength(str);
    }

    @Test
    public void test() {
        File[] classFiles = getClassFiles("./src/main/java/com/yy/easyMvc/annotation");
        for (File classFile : classFiles) {
            System.out.println(classFile);
        }
    }

}
```

DispatcherServlet:

```java
package com.yy.easyMvc.servlet;

import com.yy.easyMvc.annotation.Controller;
import com.yy.easyMvc.annotation.RequestMapping;
import com.yy.easyMvc.bean.ControllerBean;
import com.yy.easyMvc.bean.ModelAndView;
import com.yy.easyMvc.util.ClassUtil;
import javax.servlet.ServletException;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.io.IOException;
import java.lang.reflect.Method;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

/**
 * 前端控制器
 * @author ycr
 * @date 2020/6/24
 */
public class DispatchServlet extends HttpServlet{

    //
    private Map<String, ControllerBean> urlMap = new HashMap<>();

    /**
     * 初始化方法,Servlet的生命周期交由Tomcat容器管理
     * 设置了加载顺序,在容器启动的时候就初始化该Servlet
     */
    @Override
    public void init() {
        // 容器初始化就去加载该包下面贴有@Controller注解的类
        List<Class<?>> classList = ClassUtil.getClassListByAnnotation("com.yy.hello", Controller.class);
        // 将贴有@RequestMapping注解的方法封装到ControllerBean对象中
        for (Class<?> clazz : classList) {
            Method[] methods = clazz.getDeclaredMethods();
            if (methods != null && methods.length != 0) {
                for (Method method : methods) {
                    RequestMapping annotation = method.getAnnotation(RequestMapping.class);
                    if (annotation != null) {
                        String url = annotation.value();
                        urlMap.put(url, new ControllerBean(clazz, method));
                    }
                }
            }
        }

        System.out.println(urlMap);

    }

    /**
     * service方法,配置了urlMapping所有的请求都到前端控制器,所有的请求都会到该方法,
     * @param req 请求对象
     * @param resp 响应对象
     */
    @Override
    protected void service(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {

        // 获取请求的资源名称
        String requestURI = req.getRequestURI();
        if ("/favicon.ico".equals(requestURI)) {
            return;
        }

        // 去掉最后一个点后面的内容
        if (requestURI != null && requestURI.contains(".")) {
            requestURI = requestURI.substring(0, requestURI.lastIndexOf("."));
        }
        // 根据请求对象的url,寻找对应的Controller方法的包装对象
        ControllerBean controllerBean = urlMap.get(requestURI);
        if (controllerBean == null) {
//            req.getRequestDispatcher(requestURI).forward(req, resp);
            return;
        }

        // 获取当前请求对应的Controller类和对应的映射方法
        Class<?> controllerClass = controllerBean.getControllerBean();
        Method controllerMethod = controllerBean.getControllerMethod();
        try {
            Object controllerInstance = controllerClass.newInstance();
            Object o = controllerMethod.invoke(controllerInstance);
            if (o != null && o.getClass() == ModelAndView.class) {
                this.resultHandler(req, resp, (ModelAndView) o);
            }

        } catch (Exception e) {
            e.printStackTrace();
        }

    }

    /**
     * 处理响应
     * @param req 请求对象
     * @param resp 响应对象
     * @param modelAndView 视图和数据对象
     */
    private void resultHandler(HttpServletRequest req, HttpServletResponse resp, ModelAndView modelAndView) throws ServletException, IOException {
        String viewName = modelAndView.getViewName();
        Map<String, Object> model = modelAndView.getModel();
        for (Map.Entry<String, Object> entry : model.entrySet()) {
            String key = entry.getKey();
            Object value = entry.getValue();
            req.setAttribute(key, value);
        }

        String prefix = "/WEB-INF/views/";
        String suffix = ".jsp";

        req.getRequestDispatcher(prefix + viewName + suffix ).forward(req, resp);
    }
}
```
